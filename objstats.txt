# Levels with most entities of id (show n)
def maxid(id,n)
  $tabs.map{|t| Dir.entries(t).select{ |f| File.file?("#{t}/#{f}") }.map{ |f| [f, _unpack(File.binread("#{t}/#{f}")[0xB8+966+2*id..0xB8+966+2*id+1])] }}.flatten(1).sort_by{|f|-f[1]}.take(n).each_with_index.map{|f,i|"#{"%03d" % (i+1)} #{"%03d" % 
f[1]} - #{f[0]}"}.join("\n")
end

# Levels with most different kinds of entities
def maxent(id,n)
  puts $tabs.map{ |t|
    Dir.entries(t).select{ |f|
      File.file?("#{t}/#{f}")
    }.map{ |f|

      [f, amount]
    }
  }.flatten(1)
   .sort_by{ |f| -f[1]}

   .each_with_index
   .map{ |f,i| "#{"%03d" % (i+1)} #{"%03d" % f[1]} - #{f[0]}" }
   .join("\n")
end
